#include <iostream>
#include <cmath>

bool lagya(int xK, int yK, int xL, int yL) {
    return xK == xL || yK == yL;
}

bool slon(int xK, int yK, int xS, int yS) {
    return std::abs(xK - xS) == std::abs(yK - yS);
}

bool vse(int xK, int yK, int xL, int yL, int xS, int yS) {
    return lagya(xK, yK, xL, yL) && slon(xK, yK, xS, yS);
}

int main() {
    std::cout << "Введите координаты короля: " << std::endl;
    int xK, yK;
    std::cin >> xK >> yK;

    std::cout << "Введите координаты ладьи: " << std::endl;
    int xL, yL;
    std::cin >> xL >> yL;

    std::cout << "Введите координаты слона: " << std::endl;
    int xS, yS;
    std::cin >> xS >> yS;

    if (vse(xK, yK, xL, yL, xS, yS)) {
        std::cout << "Королю угрожают слон и ладья" << std::endl;
    } else if (lagya(xK, yK, xL, yL)) {
        std::cout << "Королю угрожает ладья" << std::endl;
    } else if (slon(xK, yK, xS, yS)) {
        std::cout << "Королю угрожает слон" << std::endl;
    } else {
        std::cout << "Король не находится под угрозой" << std::endl;
    }

    return 0;
}
